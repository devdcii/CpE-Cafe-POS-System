      $total_orders = $conn->query("SELECT COUNT(*) as c FROM orders")->fetch_assoc()['c'] ?? 0;
      $total_products = $conn->query("SELECT COUNT(*) as c FROM products")->fetch_assoc()['c'] ?? 0;
      $low_stock = $conn->query("SELECT COUNT(*) as c FROM products WHERE stock <= 10")->fetch_assoc()['c'] ?? 0;

      // Inventory by category
      $inv_res = $conn->query("SELECT name, category, stock FROM products ORDER BY category, name");
      $products_by_category = [];
      while ($row = $inv_res->fetch_assoc()) {
        $products_by_category[$row['category']][] = $row;
      }
    ?>

    <!-- Stats -->
    <div class="stats-grid">
      <div class="stat-card"><div class="stat-value">₱<?= number_format($total_sales, 2) ?></div><div class="stat-label">Total Sales</div></div>

      <div class="stat-card"><div class="stat-value"><?= $total_products ?></div><div class="stat-label">Total Products</div></div>
      <div class="stat-card"><div class="stat-value"><?= $low_stock ?></div><div class="stat-label">Low Stock</div></div>
    </div>

    <!-- Dashboard -->
    <div class="dashboard-grid">
      <div class="card">
        <h2>📋 Recent Orders</h2>
        <?php
        $res = $conn->query("SELECT id,total,DATE_FORMAT(created_at,'%M %d, %Y %h:%i %p') as dt FROM orders ORDER BY created_at DESC LIMIT 12");
        if ($res->num_rows) {
          echo "<table><thead><tr><th>Order #</th><th>Total</th><th>Date</th></tr></thead><tbody>";
          while ($o=$res->fetch_assoc()) {
            echo "<tr><td>#{$o['id']}</td><td>₱".number_format($o['total'],2)."</td><td>{$o['dt']}</td></tr>";
          }
          echo "</tbody></table>";
        } else echo "<div class='no-data'>No orders found</div>";
        ?>
      </div>

      <div class="card">
        <h2>🏆 Top Selling</h2>
        <?php
        $res = $conn->query("SELECT p.name,p.category,SUM(oi.quantity) as sold FROM order_items oi JOIN products p ON oi.product_id=p.id GROUP BY p.id ORDER BY sold DESC LIMIT 10");
        if ($res->num_rows) {
          echo "<table><thead><tr><th>Rank</th><th>Product</th><th>Category</th><th>Sold</th></tr></thead><tbody>";
          $r=1;
          while ($p=$res->fetch_assoc()) {
            echo "<tr><td>$r</td><td>".htmlspecialchars($p['name'])."</td><td>".htmlspecialchars($p['category'])."</td><td>{$p['sold']}</td></tr>";
            $r++;
          }
          echo "</tbody></table>";
        } else echo "<div class='no-data'>No sales data</div>";
        ?>
      </div>
    </div>

    <!-- Inventory -->
    <div class="card">
      <h2>📦 Inventory Status</h2>
      <?php if(empty($products_by_category)): ?>
        <div class='no-data'>No products found</div>
      <?php else: foreach ($products_by_category as $cat=>$prods): ?>
        <div class="category-section">
          <h3><?= htmlspecialchars($cat) ?></h3>
          <table>
            <thead><tr><th>Product</th><th>Stock</th><th>Status</th></tr></thead>
            <tbody>
              <?php foreach($prods as $p):
                $cls=$p['stock']<=10?'status-low':($p['stock']<=20?'status-medium':'status-good');

                <tr><td><?= htmlspecialchars($p['name']) ?></td><td><?= $p['stock'] ?></td><td><span class="<?= $cls ?>"><?= $txt ?></span></td></tr>

      $res = $conn->query("SELECT DATE(created_at) as d,SUM(total) as t,COUNT(*) as c FROM orders WHERE created_at>=DATE_SUB(NOW(),INTERVAL 7 DAY) GROUP BY DATE(created_at) ORDER BY d DESC");
      if ($res->num_rows) {
        echo "<table><thead><tr><th>Date</th><th>Orders</th><th>Sales</th></tr></thead><tbody>";
        while ($day=$res->fetch_assoc()) {
          echo "<tr><td>".date('M d, Y',strtotime($day['d']))."</td><td>{$day['c']}</td><td>₱".number_format($day['t'],2)."</td></tr>";
        }
        echo "</tbody></table>";
      } else echo "<div class='no-data'>No sales</div>";
      ?>
    </div>

    <?php
    } catch (Exception $e) {
